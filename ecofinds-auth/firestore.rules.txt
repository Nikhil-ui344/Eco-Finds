// Temporary Firebase rules for testing (to be used in Firebase Console)
// Navigate to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow all reads and writes for testing purposes
    // WARNING: These rules allow anyone to read/write your database
    // Only use for development/testing
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
